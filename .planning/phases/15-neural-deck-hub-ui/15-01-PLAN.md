    **4. Create `src/app/api/profile/route.ts`**: GET endpoint that loads the full PlayerProfile by sessionId query parameter. Uses `profileRepository.load(sessionId)`. If no profile found, creates a default profile using `createDefaultProfile(sessionId)`, saves it, and returns it. Returns JSON with `{ profile: PlayerProfile }`.
    ```typescript
    import { NextRequest, NextResponse } from 'next/server';
    import { profileRepository } from '@/app/persistence/fs-profile-repository';
    import { createDefaultProfile } from '@/shared/profile';

    export async function GET(req: NextRequest) {
      const sessionId = req.nextUrl.searchParams.get('sessionId');
      if (!sessionId) {
        return NextResponse.json({ error: 'Missing sessionId' }, { status: 400 });
      }
      try {
        let profile = await profileRepository.load(sessionId);
        if (!profile) {
          profile = createDefaultProfile(sessionId);
          await profileRepository.save(profile);
        }
        return NextResponse.json({ profile });
      } catch (error) {
        console.error('[Profile] Error:', error);
        return NextResponse.json({ error: 'Internal server error' }, { status: 500 });
      }
    }
    ```
