GET /api/run-mode/available?sessionId={id}
Response: { modes: [ { mode, name, description, available, reason, attemptsRemaining } ] }

<!-- Run mode launch API -->
POST /api/run-mode/launch
Body: { sessionId, mode }
Response: { success, config: { mode, seed, name } }

<!-- Leaderboard APIs -->
GET /api/leaderboard/daily → { entries: [{ playerName, score, floor, timestamp }] }
