POST /api/economy/purchase
Body: { sessionId, itemIndex, weekSeed }
Response: { success, purchased, remainingScrap }

<!-- Upgrade API -->
POST /api/economy/upgrade
Body: { sessionId, shellId, stat: 'speed'|'armor'|'stability'|'additionalPorts' }

<!-- Sell API -->
POST /api/vault/sell
Body: { sessionId, entityId }
